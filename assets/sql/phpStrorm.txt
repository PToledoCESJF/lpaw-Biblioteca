# Install PhpStorm on Linux Mint/Ubuntu Desktop
 
# PhpStorm requires JVM
 
# Download from https://www.jetbrains.com/phpstorm/download/
 
cd ~/Downloads/
# Unzip/extract .tar.gz file
tar -xvzf PhpStorm-9.0.2.tar.gz -C ~/Documents/
 
cd ~/Documents/<PhpStorm>/
# Linux Installation Instructions
vi Install-Linux-tar.txt
 
# Change dir to PhpStorm bin dir
cd ~/Documents/<PhpStorm>/bin/
 
./phpstorm.sh
